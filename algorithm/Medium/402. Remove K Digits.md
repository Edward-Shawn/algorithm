# 코드

class Solution {
public:
    string removeKdigits(string num, int k) {
       	string res = "";

	for (int i = 0; i < num.size();)
	{
		if (res.size() == 0 || res.back() <= num[i] || k == 0)
			res.push_back(num[i]), i++;
		else if (res.back() > num[i] && k != 0)
			res.pop_back(), k--;
	}

	while (k--)
		res.pop_back();
	
	while (true)
	{
		if (res.front() == '0')
			res.erase(res.begin());
		else
			break;
	}

	return res.size() == 0 ? "0" : res;
    }
};
