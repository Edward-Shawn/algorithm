# 코드 리뷰 
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
    int min =INT_MAX, max = 0, sum = 0;
	for (int i = 0; i < prices.size(); i++)
	{
        if (max > prices[i])
		{
			sum += max - min;
			min = prices[i];
			max = 0;
		}
		else if (min > prices[i])
		{
			min = prices[i];
			max = 0;
		}
        else if (max < prices[i])
			max = prices[i];
		
	}

	return max != 0 ? sum + max - min : sum;
  }
};
```
