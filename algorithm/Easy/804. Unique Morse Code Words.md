804. Unique Morse Code Words

# 코드

```cpp
class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        	string mose[26]{ ".-", "-...", "-.-.", "-..", ".", "..-.", "--.", "....", "..", ".---", "-.-", ".-..", "--", "-.", "---", ".--.", "--.-", ".-.", "...", "-", "..-", "...-", ".--", "-..-", "-.--", "--.." };
	map<string, bool> maplist = map<string, bool>();
	for (int i = 0; i < words.size(); i++)
	{
		string str = "";
		for (int j = 0; j < words[i].size(); j++)
			str.append(mose[words[i][j] - 'a']);
		maplist[str] = true;
	}
	return maplist.size();
    }
};
```
