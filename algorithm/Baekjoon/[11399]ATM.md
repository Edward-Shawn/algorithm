# 코드

```cpp

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n,  res = 0 , sum = 0;
	cin >> n;
	vector<int> vec = vector<int>(n,0);
	for (int i = 0; i < n; i++)
		cin >> vec[i];
	
	sort(vec.begin(), vec.end());
	for (int i = 0; i < vec.size(); i++)
		res += sum += vec[i];
	
	cout << res << endl;
	return 0;
}
```
