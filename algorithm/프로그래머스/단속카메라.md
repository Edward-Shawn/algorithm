# 코드

```cpp
#include <string>
#include <vector>

using namespace std;
#define INT_MAX  2147483647
int solution(vector<vector<int>> routes) {
   int answer = 0;

	while (true)
	{
		if (routes.size() == 0)
			return answer;
		int m = INT_MAX;
		for (int i = 0; i < routes.size(); i++)
			m = min(routes[i][1], m);
        answer++;
		for (int i = routes.size() - 1; i > -1; i--)
			if (routes[i][1] >= m && routes[i][0] <= m)
				routes.erase(routes.begin() + i);
	}
}
```
